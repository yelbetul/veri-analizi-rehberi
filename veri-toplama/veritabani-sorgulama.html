Advanced SQL Pattern Demo
-- 1. Window Functions - Satış Analizi
SELECT
salesperson_id,
sale_date,
amount,
SUM(amount) OVER(
PARTITION BY salesperson_id
ORDER BY sale_date
ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
) AS running_total,
AVG(amount) OVER(
PARTITION BY salesperson_id
ORDER BY sale_date
ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
) AS moving_avg_3_days,
LAG(amount, 1) OVER(
PARTITION BY salesperson_id
ORDER BY sale_date
) AS previous_sale
FROM sales
ORDER BY salesperson_id, sale_date;

-- 2. Recursive CTE - Organizasyon Şeması
WITH RECURSIVE org_hierarchy AS (
-- Anchor: En üst seviye yöneticiler
SELECT
employee_id,
manager_id,
name,
title,
1 as level,
ARRAY[employee_id] as path
FROM employees
WHERE manager_id IS NULL

UNION ALL

-- Recursive: Alt çalışanlar
SELECT
e.employee_id,
e.manager_id,
e.name,
e.title,
oh.level + 1,
oh.path || e.employee_id
FROM employees e
JOIN org_hierarchy oh ON e.manager_id = oh.employee_id
)
SELECT
LPAD('', (level - 1) * 4, ' ') || name as indented_name,
title,
level
FROM org_hierarchy
ORDER BY path;

-- 3. Gaps and Islands - Eksik Veri Tespiti
WITH numbered_sales AS (
SELECT
sale_date,
amount,
ROW_NUMBER() OVER (ORDER BY sale_date) as rn,
sale_date - INTERVAL '1 day' *
(ROW_NUMBER() OVER (ORDER BY sale_date) - 1) as group_id
FROM sales
WHERE sale_date BETWEEN '2024-01-01' AND '2024-01-31'
),
date_groups AS (
SELECT
MIN(sale_date) as island_start,
MAX(sale_date) as island_end,
COUNT(*) as consecutive_days
FROM numbered_sales
GROUP BY group_id
)
SELECT
island_start,
island_end,
consecutive_days
FROM date_groups
WHERE consecutive_days >= 3 -- 3+ ardışık gün
ORDER BY island_start;

-- 4. Pivot Query - Aylık Satış Özeti
SELECT
product_category,
SUM(CASE WHEN EXTRACT(MONTH FROM sale_date) = 1 THEN amount ELSE 0 END) as jan,
SUM(CASE WHEN EXTRACT(MONTH FROM sale_date) = 2 THEN amount ELSE 0 END) as feb,
SUM(CASE WHEN EXTRACT(MONTH FROM sale_date) = 3 THEN amount ELSE 0 END) as mar,
SUM(CASE WHEN EXTRACT(MONTH FROM sale_date) = 4 THEN amount ELSE 0 END) as apr,
SUM(amount) as total
FROM sales s
JOIN products p ON s.product_id = p.product_id
WHERE EXTRACT(YEAR FROM sale_date) = 2024
GROUP BY product_category
ORDER BY total DESC;

Query Optimization

Indexing strategy, query performance analysis ve optimizasyon teknikleri

🚀 Ne İşe Yarar?

Performans artışı Index optimizasyonu Query analizi Database tuning

Kullanım Amacı: SQL sorgularının performansını artırmak, index kullanımını optimize etmek ve database kaynak kullanımını iyileştirmek. Örnek: EXPLAIN ANALYZE ile query execution plan'ını analiz etmek.

💡 Ne Zaman Kullanılır?

Yavaş çalışan sorguları optimize etmek, büyük veri setlerinde arama performansını artırmak ve database kaynak kullanımını iyileştirmek için.

Index Türleri

  • B-Tree Index
    • Eşitlik ve aralık sorguları
    • Sıralama operasyonları
    • Default index türü
  • Hash Index
    • Sadece eşitlik sorguları
    • Çok hızlı lookup
    • Memory tabloları
  • Bitmap Index
    • Düşük cardinality
    • Read-heavy workload
    • Data warehouse
  • Full-Text Index
    • Text aramaları
    • Relevance scoring
    • Linguistik analiz

Index Stratejileri

Composite Index

• WHERE clause sırası
• Covering index
• Column selectivity

Partial Index

• WHERE koşulu ile
• Daha küçük index
• Sorgu pattern'ine göre

Expression Index

• Fonksiyon sonuçları
• Lowercase aramalar
• Date extraction

Index Optimizasyon Demo
-- 1. Mevcut Index'leri Analiz Et
SELECT
schemaname,
tablename,
indexname,
indexdef
FROM pg_indexes
WHERE schemaname = 'public'
ORDER BY tablename, indexname;

-- 2. Index Kullanım İstatistikleri
SELECT
schemaname,
tablename,
indexname,
idx_scan as index_scans,
idx_tup_read as tuples_read,
idx_tup_fetch as tuples_fetched
FROM pg_stat_user_indexes
WHERE schemaname = 'public'
ORDER BY idx_scan DESC;

-- 3. Etkili Composite Index Oluşturma
-- Kötü: WHERE sırasına uyumsuz
CREATE INDEX idx_sales_bad ON sales (sale_date, salesperson_id, product_id);

-- İyi: WHERE clause sırasına uyumlu
CREATE INDEX idx_sales_optimized ON sales (salesperson_id, sale_date, product_id);

-- 4. Partial Index - Aktif Kullanıcılar
CREATE INDEX idx_users_active ON users (last_login_date)
WHERE status = 'active';

-- 5. Expression Index - Case Insensitive Arama
CREATE INDEX idx_users_email_lower ON users (LOWER(email));

-- 6. Covering Index - Extra Column Include
CREATE INDEX idx_orders_covering ON orders (customer_id, order_date)
INCLUDE (total_amount, status);

-- 7. Index Kullanımını Zorlama (PostgreSQL)
SET enable_seqscan = off;
EXPLAIN ANALYZE
SELECT * FROM sales
WHERE salesperson_id = 123
AND sale_date BETWEEN '2024-01-01' AND '2024-01-31';

-- 8. Index Boyut Analizi
SELECT
indexname,
pg_size_pretty(pg_relation_size(indexname::regclass)) as size
FROM pg_indexes
WHERE schemaname = 'public'
ORDER BY pg_relation_size(indexname::regclass) DESC;

NoSQL Data Extraction

MongoDB, Cassandra, Redis ve diğer NoSQL veritabanlarından veri çıkarma teknikleri

🚀 Ne İşe Yarar?

MongoDB Aggregation Cassandra Queries Redis Data Structures Elasticsearch DSL

Kullanım Amacı: Yapılandırılmamış ve yarı yapılandırılmış verilerden bilgi çıkarmak, NoSQL veritabanlarında kompleks sorgular çalıştırmak. Örnek: MongoDB aggregation pipeline ile e-ticaret verilerini analiz etmek.

💡 Ne Zaman Kullanılır?

MongoDB'de kompleks veri işleme, gruplama, filtreleme ve dönüştürme işlemleri gerektiğinde, SQL'deki JOIN ve GROUP BY benzeri işlemler yapmak için.

Aggregation Stages

  • $match
    • WHERE clause benzeri
    • Document filtreleme
    • Index kullanabilir
  • $group
    • GROUP BY benzeri
    • Aggregation operatörleri
    • _id ile gruplama
  • $lookup
    • JOIN işlemi
    • Collection birleştirme
    • Left outer join
  • $project
    • SELECT benzeri
    • Field seçimi
    • Yeni field oluşturma

Aggregation Operatörleri

Matematiksel

• $sum, $avg, $min, $max
• $multiply, $divide
• $sqrt, $log

Diziler

• $push, $addToSet
• $slice, $size
• $filter, $map

String

• $concat, $substr
• $toLower, $toUpper
• $split, $trim

MongoDB Aggregation Demo
// 1. E-ticaret Sipariş Analizi
db.orders.aggregate([
{
$match: {
orderDate: {
$gte: ISODate("2024-01-01"),
$lt: ISODate("2024-02-01")
},
status: "completed"
}
},
{
$lookup: {
from: "customers",
localField: "customerId",
foreignField: "_id",
as: "customer"
}
},
{ $unwind: "$customer" },
{
$unwind: "$items"
},
{
$lookup: {
from: "products",
localField: "items.productId",
foreignField: "_id",
as: "product"
}
},
{ $unwind: "$product" },
{
$group: {
_id: {
category: "$product.category",
month: { $month: "$orderDate" }
},
totalSales: { $sum: "$items.quantity" },
totalRevenue: {
$sum: {
$multiply: ["$items.quantity", "$items.unitPrice"]
}
},
averageOrderValue: { $avg: "$totalAmount" },
uniqueCustomers: { $addToSet: "$customerId" }
}
},
{
$project: {
category: "$_id.category",
month: "$_id.month",
totalSales: 1,
totalRevenue: 1,
averageOrderValue: 1,
customerCount: { $size: "$uniqueCustomers" },
revenuePerCustomer: {
$divide: ["$totalRevenue", { $size: "$uniqueCustomers" }]
}
}
},
{
$sort: { totalRevenue: -1 }
}
]);

// 2. Kullanıcı Davranış Analizi
db.userSessions.aggregate([
{
$match: {
timestamp: {
$gte: ISODate("2024-01-01"),
$lt: ISODate("2024-02-01")
}
}
},
{
$group: {
_id: "$userId",
totalSessions: { $sum: 1 },
totalDuration: { $sum: "$duration" },
pagesVisited: { $push: "$pageUrl" },
firstSession: { $min: "$timestamp" },
lastSession: { $max: "$timestamp" }
}
},
{
$project: {
userId: "$_id",
totalSessions: 1,
averageDuration: { $divide: ["$totalDuration", "$totalSessions"] },
uniquePages: { $size: { $setUnion: ["$pagesVisited", []] } },
sessionFrequency: {
$divide: [
{ $subtract: ["$lastSession", "$firstSession"] },
{ $multiply: ["$totalSessions", 1000 * 60 * 60 * 24] }
]
}
}
},
{
$bucket: {
groupBy: "$totalSessions",
boundaries: [1, 3, 10, 30, 100],
default: "frequent",
output: {
count: { $sum: 1 },
avgDuration: { $avg: "$averageDuration" },
avgPages: { $avg: "$uniquePages" }
}
}
}
]);

// 3. Real-time Analytics Pipeline
db.analytics.aggregate([
{
$match: {
timestamp: {
$gte: ISODate("2024-01-15T00:00:00Z"),
$lt: ISODate("2024-01-15T23:59:59Z")
}
}
},
{
$group: {
_id: {
$dateToString: {
format: "%Y-%m-%dT%H:00:00",
date: "$timestamp"
}
},
pageViews: {
$sum: { $cond: [{ $eq: ["$eventType", "pageView"] }, 1, 0] }
},
clicks: {
$sum: { $cond: [{ $eq: ["$eventType", "click"] }, 1, 0] }
},
conversions: {
$sum: { $cond: [{ $eq: ["$eventType", "conversion"] }, 1, 0] }
},
uniqueUsers: { $addToSet: "$userId" }
}
},
{
$project: {
hour: "$_id",
pageViews: 1,
clicks: 1,
conversions: 1,
uniqueUsers: { $size: "$uniqueUsers" },
conversionRate: {
$cond: [
{ $eq: ["$pageViews", 0] },
0,
{ $divide: ["$conversions", "$pageViews"] }
]
}
}
},
{ $sort: { hour: 1 } }
]);

Performance Analysis Demo
-- 1. EXPLAIN ANALYZE ile Query Analizi
EXPLAIN (ANALYZE, BUFFERS, FORMAT JSON)
SELECT
c.customer_name,
COUNT(o.order_id) as order_count,
SUM(o.total_amount) as total_spent
FROM customers c
JOIN orders o ON c.customer_id = o.customer_id
WHERE o.order_date >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY c.customer_id, c.customer_name
HAVING SUM(o.total_amount) > 1000
ORDER BY total_spent DESC
LIMIT 10;

-- 2. pg_stat_statements ile Yavaş Sorgular
SELECT
query,
calls,
total_exec_time,
mean_exec_time,
rows,
100.0 * shared_blks_hit / nullif(shared_blks_hit + shared_blks_read, 0) AS hit_percent
FROM pg_stat_statements
ORDER BY mean_exec_time DESC
LIMIT 10;

-- 3. Index Kullanım İstatistikleri
SELECT
schemaname,
tablename,
indexname,
idx_scan as index_scans,
idx_tup_read as tuples_read,
idx_tup_fetch as tuples_fetched,
pg_size_pretty(pg_relation_size(indexrelid)) as index_size
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC;

-- 4. Table İstatistikleri ve Bloat Analizi
SELECT
schemaname,
tablename,
seq_scan,
seq_tup_read,
idx_scan,
idx_tup_fetch,
n_tup_ins,
n_tup_upd,
n_tup_del,
n_live_tup,
n_dead_tup,
round(n_dead_tup::numeric / (n_live_tup + n_dead_tup) * 100, 2) as dead_tup_percent
FROM pg_stat_user_tables
WHERE (n_live_tup + n_dead_tup) > 0
ORDER BY dead_tup_percent DESC;

-- 5. Connection Pool Monitoring
SELECT
datname,
usename,
application_name,
client_addr,
state,
query_start,
query
FROM pg_stat_activity
WHERE state = 'active'
AND datname = current_database();

-- 6. Lock Monitoring
SELECT
relation::regclass as table_name,
locktype,
mode,
granted,
pg_blocking_pids(pid) as blocking_pids
FROM pg_locks
WHERE relation IS NOT NULL
AND NOT granted;

-- 7. Table Size ve Bloat Analysis
SELECT
schemaname,
tablename,
pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) as total_size,
pg_size_pretty(pg_relation_size(schemaname||'.'||tablename)) as table_size,
pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename) -
pg_relation_size(schemaname||'.'||tablename)) as index_size
FROM pg_tables
WHERE schemaname NOT IN ('pg_catalog', 'information_schema')
ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC
LIMIT 20;
