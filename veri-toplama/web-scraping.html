import requests
from bs4 import BeautifulSoup

def scrape_news_site(url):
  try:
    # Sayfayƒ± indir
    response = requests.get(url)
    response.raise_for_status() # HTTP hatalarƒ±nƒ± kontrol et
    
    # BeautifulSoup ile parse et
    soup = BeautifulSoup(response.content, 'lxml')
    
    # Haberleri bul
    news_items = []
    articles = soup.find_all('article', class_='news-item')
    
    for article in articles:
      # √áoklu selector stratejisi
      title_elem = article.find('h2', class_='title') or article.find('h2')
      title = title_elem.get_text(strip=True) if title_elem else 'Ba≈ülƒ±k bulunamadƒ±'
      
      link_elem = title_elem.find('a') if title_elem else None
      link = link_elem['href'] if link_elem and link_elem.has_attr('href') else ''
      
      date_elem = article.find('span', class_='date')
      date = date_elem.get_text(strip=True) if date_elem else 'Tarih bulunamadƒ±'
      
      summary_elem = article.find('p', class_='summary')
      summary = summary_elem.get_text(strip=True) if summary_elem else '√ñzet bulunamadƒ±'
      
      news_items.append({
        'title': title,
        'link': link,
        'date': date,
        'summary': summary
      })
    
    return news_items
    
  except requests.RequestException as e:
    print(f"ƒ∞stek hatasƒ±: {e}")
    return []
  except Exception as e:
    print(f"Parse hatasƒ±: {e}")
    return []

import requests
from bs4 import BeautifulSoup
import time

def scrape_paginated_site(base_url, max_pages=10):
  all_data = []
  current_page = 1
  
  while current_page <= max_pages:
    try:
      # URL olu≈ütur
      if '?' in base_url:
        url = f"{base_url}&page={current_page}"
      else:
        url = f"{base_url}?page={current_page}"
      
      print(f"Sayfa {current_page} √ßekiliyor: {url}")
      
      response = requests.get(url)
      response.raise_for_status()
      
      soup = BeautifulSoup(response.content, 'lxml')
      
      # Verileri √ßƒ±kar
      page_data = extract_data_from_page(soup)
      
      # Eƒüer sayfada veri yoksa d√∂ng√ºy√º sonlandƒ±r
      if not page_data:
        print("Sayfada veri bulunamadƒ±, scraping sonlandƒ±rƒ±lƒ±yor.")
        break
      
      all_data.extend(page_data)
      
      # Sonraki sayfa kontrol√º

        print("Son sayfaya ula≈üƒ±ldƒ±.")
        break

    except requests.RequestException as e:
      print(f"Sayfa {current_page} i√ßin hata: {e}")
      break
    
  return all_data

def extract_data_from_page(soup):
  # Sayfadan veri √ßƒ±karma mantƒ±ƒüƒ±
  items = []
  # √ñrnek: √úr√ºn kartlarƒ±nƒ± bul
  product_cards = soup.find_all('div', class_='product-card')
  
  for card in product_cards:

    price_elem = card.find('span', class_='price')
    
    if title_elem and price_elem:
      items.append({
        'title': title_elem.get_text(strip=True),
        'price': price_elem.get_text(strip=True)
      })
  
  return items

def has_next_page(soup):
  # Sonraki sayfa kontrol√º
  next_button = soup.find('a', class_='next-page')
  return next_button is not None and 'disabled' not in next_button.get('class', [])

def setup_driver(headless=True):
  """Chrome WebDriver kurulumu"""

  if headless:

  driver = webdriver.Chrome(options=chrome_options)
  driver.execute_script("Object.defineProperty(navigator, 'webdriver', {get: () => undefined})")
  
  return driver

def scrape_dynamic_content(url, wait_time=10):
  """Dinamik i√ßerik scraping fonksiyonu"""
  driver = setup_driver(headless=True)
  scraped_data = []
  
  try:

def scroll_and_collect(driver, data_list, max_scrolls=10):
  """Infinite scroll handling ve veri toplama"""
  last_height = driver.execute_script("return document.body.scrollHeight")
  scroll_attempts = 0
  
  while scroll_attempts < max_scrolls:
    # Sayfayƒ± en alta kaydƒ±r
    driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
    time.sleep(2) # Yeni i√ßeriklerin y√ºklenmesini bekle
    
    # Yeni i√ßerikleri topla
    new_items = extract_items_from_page(driver)
    for item in new_items:
      if item not in data_list:
        data_list.append(item)
    
    # Yeni y√ºkseklik kontrol√º
    new_height = driver.execute_script("return document.body.scrollHeight")
    if new_height == last_height:
      print("Daha fazla i√ßerik yok, scrolling durduruluyor.")
      break
    
    last_height = new_height
    scroll_attempts += 1
    print(f"Scroll {scroll_attempts} tamamlandƒ±. Toplam {len(data_list)} √∂ƒüe bulundu.")

def extract_items_from_page(driver):
  """Sayfadan √∂ƒüeleri √ßƒ±kar"""
  items = []
  try:
    # √ñrnek: Sosyal medya g√∂nderilerini √ßƒ±kar

        time_elem = post.find_element(By.CSS_SELECTOR, ".time, .date, .timestamp")

        if text: # Bo≈ü olmayan g√∂nderileri ekle
          items.append({
            'text': text,
            'timestamp': timestamp
          })
      except Exception as e:
        continue # Hatalƒ± √∂ƒüeleri atla
    
  except Exception as e:
    print(f"√ñƒüe √ßƒ±karma hatasƒ±: {e}")
  
  return items

class EcommerceSpider(scrapy.Spider):
  name = 'ecommerce_spider'
  allowed_domains = ['example-ecommerce.com']
  start_urls = ['https://example-ecommerce.com/products']
  
  custom_settings = {
    'CONCURRENT_REQUESTS': 4,
    'DOWNLOAD_DELAY': 1,
    'USER_AGENT': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
    'ROBOTSTXT_OBEY': True,
    'FEED_FORMAT': 'json',
    'FEED_URI': 'products_%(time)s.json'
  }
  
  def parse(self, response):
    """Ana sayfa parsing"""
    # Kategori linklerini bul
    categories = response.css('.category-list a::attr(href)').getall()
    
    for category_url in categories:
      yield response.follow(
        category_url,
        callback=self.parse_category,
        meta={'category': category_url.split('/')[-1]}
      )
  
  def parse_category(self, response):
    """Kategori sayfasƒ± parsing"""
    category = response.meta.get('category', 'unknown')
    
    # √úr√ºn kartlarƒ±nƒ± i≈üle
    products = response.css('.product-card')
    
    for product in products:
      yield {
        'category': category,
        'title': product.css('.product-title::text').get().strip(),
        'price': product.css('.price::text').get().strip(),
        'url': response.urljoin(product.css('a::attr(href)').get()),
        'scraped_at': datetime.now().isoformat()
      }
    
    # Sayfalama - sonraki sayfa
    next_page = response.css('.next-page::attr(href)').get()
    if next_page:
      yield response.follow(
        next_page,
        callback=self.parse_category,
        meta={'category': category}
      )
