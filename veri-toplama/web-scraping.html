Static Content Extraction

Statik HTML içeriğinden veri çıkarma teknikleri ve optimizasyon stratejileri

🚀 Ne İşe Yarar?

Basit web siteleri Hızlı veri çekme Düşük kaynak kullanımı Kolay implementasyon

Kullanım Amacı: JavaScript gerektirmeyen, statik HTML içerikli web sitelerinden veri çıkarmak. Örnek: Haber sitelerinden makale başlıklarını ve içeriklerini çekmek.

💡 Ne Zaman Kullanılır?

HTML parsing ihtiyacı olduğunda, hızlı prototipleme için, küçük ölçekli scraping projelerinde.

Selector Türleri

  • • CSS Selectors - find(), select()
  • • Tag Name - find_all('tag')
  • • Attributes - find(attrs={'class': 'value'})
  • • Text Content - find(text='pattern')
  • • Regular Expressions - find_all(text=re.compile('pattern'))

En İyi Uygulamalar

  • • Robust Selectors - Birden fazla selector kullan
  • • Error Handling - try/except blokları
  • • Performance - LXML parser kullanımı
  • • Maintainability - Configurable selectors

import requests
from bs4 import BeautifulSoup
import json

def scrape_news_site(url):
  try:
    # Sayfayı indir
    response = requests.get(url)
    response.raise_for_status() # HTTP hatalarını kontrol et
    
    # BeautifulSoup ile parse et
    soup = BeautifulSoup(response.content, 'lxml')
    
    # Haberleri bul
    news_items = []
    articles = soup.find_all('article', class_='news-item')
    
    for article in articles:
      # Çoklu selector stratejisi
      title_elem = article.find('h2', class_='title') or article.find('h2')
      title = title_elem.get_text(strip=True) if title_elem else 'Başlık bulunamadı'
      
      link_elem = title_elem.find('a') if title_elem else None
      link = link_elem['href'] if link_elem and link_elem.has_attr('href') else ''
      
      date_elem = article.find('span', class_='date')
      date = date_elem.get_text(strip=True) if date_elem else 'Tarih bulunamadı'
      
      summary_elem = article.find('p', class_='summary')
      summary = summary_elem.get_text(strip=True) if summary_elem else 'Özet bulunamadı'
      
      news_items.append({
        'title': title,
        'link': link,
        'date': date,
        'summary': summary
      })
    
    return news_items
    
  except requests.RequestException as e:
    print(f"İstek hatası: {e}")
    return []
  except Exception as e:
    print(f"Parse hatası: {e}")
    return []

# Kullanım
news_url = "https://example-news-site.com"
news_data = scrape_news_site(news_url)

print(f"Toplam {len(news_data)} haber bulundu:")
for i, news in enumerate(news_data, 1):
  print(f"{i}. {news['title']} - {news['date']}")
  print(f" Özet: {news['summary'][:100]}...")
  print()

Çok sayfalı içeriğe sahip web sitelerinden tüm verileri çıkarmak için: e-ticaret ürün listeleri, blog sayfaları, forumlar.

Pagination Pattern'leri

  • Page Number Pattern
    • URL: /page/1, /page/2
    • Örnek: Blog sayfaları
  • Offset Pattern
    • URL: ?start=0, ?start=20
    • Örnek: Arama sonuçları
  • Next Button Pattern
    • "Sonraki" butonu takibi
    • Örnek: E-ticaret siteleri
  • Infinite Scroll
    • AJAX istekleri analizi
    • Örnek: Sosyal medya

Implementasyon Stratejileri

Basit Sayfa Numaralandırma

• Sabit URL pattern
• Maksimum sayfa kontrolü
• Hata toleransı

Dinamik Next Button

Pagination Handling Demo
import requests
from bs4 import BeautifulSoup
import time

def scrape_paginated_site(base_url, max_pages=10):
  all_data = []
  current_page = 1
  
  while current_page <= max_pages:
    try:
      # URL oluştur
      if '?' in base_url:
        url = f"{base_url}&page={current_page}"
      else:
        url = f"{base_url}?page={current_page}"
      
      print(f"Sayfa {current_page} çekiliyor: {url}")
      
      response = requests.get(url)
      response.raise_for_status()
      
      soup = BeautifulSoup(response.content, 'lxml')
      
      # Verileri çıkar
      page_data = extract_data_from_page(soup)
      
      # Eğer sayfada veri yoksa döngüyü sonlandır
      if not page_data:
        print("Sayfada veri bulunamadı, scraping sonlandırılıyor.")
        break
      
      all_data.extend(page_data)
      
      # Sonraki sayfa kontrolü
      if not has_next_page(soup):
        print("Son sayfaya ulaşıldı.")
        break
      
      current_page += 1
      time.sleep(1) # Politika: 1 saniye bekle
      
    except requests.RequestException as e:
      print(f"Sayfa {current_page} için hata: {e}")
      break
    
  return all_data

def extract_data_from_page(soup):
  # Sayfadan veri çıkarma mantığı
  items = []
  # Örnek: Ürün kartlarını bul
  product_cards = soup.find_all('div', class_='product-card')
  
  for card in product_cards:
    title_elem = card.find('h3', class_='product-title')
    price_elem = card.find('span', class_='price')
    
    if title_elem and price_elem:
      items.append({
        'title': title_elem.get_text(strip=True),
        'price': price_elem.get_text(strip=True)
      })
  
  return items

def has_next_page(soup):
  # Sonraki sayfa kontrolü
  next_button = soup.find('a', class_='next-page')
  return next_button is not None and 'disabled' not in next_button.get('class', [])

# Kullanım
base_url = "https://example-ecommerce.com/products"
all_products = scrape_paginated_site(base_url, max_pages=5)
print(f"Toplam {len(all_products)} ürün bulundu.")

Selenium Demo - Dinamik İçerik Scraping
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.chrome.options import Options
import time
import json

def setup_driver(headless=True):
  """Chrome WebDriver kurulumu"""
  chrome_options = Options()
  
  if headless:
    chrome_options.add_argument("--headless")
  
  # Anti-detection ayarları
  chrome_options.add_argument("--no-sandbox")
  chrome_options.add_argument("--disable-dev-shm-usage")
  chrome_options.add_argument("--disable-blink-features=AutomationControlled")
  chrome_options.add_experimental_option("excludeSwitches", ["enable-automation"])
  chrome_options.add_experimental_option('useAutomationExtension', False)
  
  # User-agent ayarı
  chrome_options.add_argument("--user-agent=Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36")
  
  driver = webdriver.Chrome(options=chrome_options)
  driver.execute_script("Object.defineProperty(navigator, 'webdriver', {get: () => undefined})")
  
  return driver

def scrape_dynamic_content(url, wait_time=10):
  """Dinamik içerik scraping fonksiyonu"""
  driver = setup_driver(headless=True)
  scraped_data = []
  
  try:
    print(f"Sayfa yükleniyor: {url}")
    driver.get(url)
    
    # Sayfanın yüklenmesini bekle
    WebDriverWait(driver, wait_time).until(
      EC.presence_of_element_located((By.TAG_NAME, "body"))
    )
    
    # Infinite scroll handling
    print("Infinite scroll işleniyor...")
    scroll_and_collect(driver, scraped_data)
    
    return scraped_data
    
  except Exception as e:
    print(f"Hata oluştu: {e}")
    return scraped_data
  finally:
    driver.quit()

def scroll_and_collect(driver, data_list, max_scrolls=10):
  """Infinite scroll handling ve veri toplama"""
  last_height = driver.execute_script("return document.body.scrollHeight")
  scroll_attempts = 0
  
  while scroll_attempts < max_scrolls:
    # Sayfayı en alta kaydır
    driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
    time.sleep(2) # Yeni içeriklerin yüklenmesini bekle
    
    # Yeni içerikleri topla
    new_items = extract_items_from_page(driver)
    for item in new_items:
      if item not in data_list:
        data_list.append(item)
    
    # Yeni yükseklik kontrolü
    new_height = driver.execute_script("return document.body.scrollHeight")
    if new_height == last_height:
      print("Daha fazla içerik yok, scrolling durduruluyor.")
      break
    
    last_height = new_height
    scroll_attempts += 1
    print(f"Scroll {scroll_attempts} tamamlandı. Toplam {len(data_list)} öğe bulundu.")

def extract_items_from_page(driver):
  """Sayfadan öğeleri çıkar"""
  items = []
  try:
    # Örnek: Sosyal medya gönderilerini çıkar
    posts = driver.find_elements(By.CSS_SELECTOR, ".post, .tweet, .card")
    
    for post in posts:
      try:
        text_elem = post.find_element(By.CSS_SELECTOR, ".content, .text, .message")
        text = text_elem.text.strip() if text_elem else ""
        
        time_elem = post.find_element(By.CSS_SELECTOR, ".time, .date, .timestamp")
        timestamp = time_elem.text.strip() if time_elem else ""
        
        if text: # Boş olmayan gönderileri ekle
          items.append({
            'text': text,
            'timestamp': timestamp
          })
      except Exception as e:
        continue # Hatalı öğeleri atla
    
  except Exception as e:
    print(f"Öğe çıkarma hatası: {e}")
  
  return items

# Kullanım örneği
if __name__ == "__main__":
  target_url = "https://example-social-media.com/feed"
  data = scrape_dynamic_content(target_url)
  
  print(f"Toplam {len(data)} gönderi toplandı:")
  for i, item in enumerate(data[:5], 1): # İlk 5 göster
    print(f"{i}. {item['timestamp']}: {item['text'][:100]}...")
  
  # JSON olarak kaydet
  with open('scraped_data.json', 'w', encoding='utf-8') as f:
    json.dump(data, f, ensure_ascii=False, indent=2)
  print("Veriler 'scraped_data.json' dosyasına kaydedildi.")

Scrapy Spider Örneği - E-ticaret Sitesi
import scrapy
from scrapy.crawler import CrawlerProcess
from scrapy.utils.project import get_project_settings
import json
from datetime import datetime

class EcommerceSpider(scrapy.Spider):
  name = 'ecommerce_spider'
  allowed_domains = ['example-ecommerce.com']
  start_urls = ['https://example-ecommerce.com/products']
  
  custom_settings = {
    'CONCURRENT_REQUESTS': 4,
    'DOWNLOAD_DELAY': 1,
    'USER_AGENT': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
    'ROBOTSTXT_OBEY': True,
    'FEED_FORMAT': 'json',
    'FEED_URI': 'products_%(time)s.json'
  }
  
  def parse(self, response):
    """Ana sayfa parsing"""
    # Kategori linklerini bul
    categories = response.css('.category-list a::attr(href)').getall()
    
    for category_url in categories:
      yield response.follow(
        category_url,
        callback=self.parse_category,
        meta={'category': category_url.split('/')[-1]}
      )
  
  def parse_category(self, response):
    """Kategori sayfası parsing"""
    category = response.meta.get('category', 'unknown')
    
    # Ürün kartlarını işle
    products = response.css('.product-card')
    
    for product in products:
      yield {
        'category': category,
        'title': product.css('.product-title::text').get().strip(),
        'price': product.css('.price::text').get().strip(),
        'url': response.urljoin(product.css('a::attr(href)').get()),
        'scraped_at': datetime.now().isoformat()
      }
    
    # Sayfalama - sonraki sayfa
    next_page = response.css('.next-page::attr(href)').get()
    if next_page:
      yield response.follow(
        next_page,
        callback=self.parse_category,
        meta={'category': category}
      )

class PriceValidationPipeline:
  """Fiyat validasyon pipeline'ı"""
  
  def process_item(self, item, spider):
    # Fiyat formatını temizle
    if 'price' in item:
      price = item['price']
      # Sadece sayısal karakterleri ve nokta tut
      cleaned_price = ''.join(c for c in price if c.isdigit() or c == '.' or c == ',')
      item['price_cleaned'] = cleaned_price
    
    return item

# Spider'ı çalıştır
if __name__ == "__main__":
  process = CrawlerProcess(get_project_settings())
  process.crawl(EcommerceSpider)
  process.start()
