// Redis ile Sliding Window Rate Limiting
const redis = require('redis');
const client = redis.createClient();

class RateLimiter {
  constructor({ windowSize, maxRequests }) {
    this.windowSize = windowSize; // saniye cinsinden
    this.maxRequests = maxRequests;
  }

  async checkRateLimit(identifier) {
    const key = `rate_limit:${identifier}`;
    const now = Date.now();
    const windowStart = now - (this.windowSize * 1000);

    try {
      // Eski kayƒ±tlarƒ± temizle
      await client.zremrangebyscore(key, 0, windowStart);
      
      // Mevcut request sayƒ±sƒ±nƒ± al
      const requestCount = await client.zcard(key);
      
      if (requestCount >= this.maxRequests) {
        return {
          allowed: false,
          remaining: 0,
          retryAfter: await this.getRetryAfter(key, windowStart)
        };
      }

      // Yeni request'i ekle
      await client.zadd(key, now, now.toString());
      await client.expire(key, this.windowSize);

      return {
        allowed: true,
        remaining: this.maxRequests - requestCount - 1,
        retryAfter: null
      };
      
    } catch (error) {
      console.error('Rate limit error:', error);
      // Hata durumunda rate limiting'i bypass et
      return { allowed: true, remaining: this.maxRequests, retryAfter: null };
    }
  }

  async getRetryAfter(key, windowStart) {
    const oldestRequest = await client.zrange(key, 0, 0, 'WITHSCORES');
    if (oldestRequest.length > 0) {
      const oldestTimestamp = parseInt(oldestRequest[1]);
      return Math.ceil((oldestTimestamp + (this.windowSize * 1000) - Date.now()) / 1000);
    }
    return this.windowSize;
  }
}

// Rate Limiting Middleware
const rateLimiter = new RateLimiter({
  windowSize: 60, // 60 saniye
  maxRequests: 100 // maksimum 100 istek
});

app.use(async (req, res, next) => {
  const identifier = req.ip; // IP bazlƒ± rate limiting
  
  const limitResult = await rateLimiter.checkRateLimit(identifier);
  
  if (!limitResult.allowed) {
    res.set({
      'X-RateLimit-Limit': rateLimiter.maxRequests,
      'X-RateLimit-Remaining': limitResult.remaining,
      'X-RateLimit-Reset': limitResult.retryAfter,
      'Retry-After': limitResult.retryAfter
    });
    return res.status(429).json({
      error: 'Too Many Requests',
      message: `Rate limit exceeded. Try again in ${limitResult.retryAfter} seconds.`,
      retryAfter: limitResult.retryAfter
    });
  }

  res.set({
    'X-RateLimit-Limit': rateLimiter.maxRequests,
    'X-RateLimit-Remaining': limitResult.remaining,
    'X-RateLimit-Reset': limitResult.retryAfter || rateLimiter.windowSize
  });

  next();
});
