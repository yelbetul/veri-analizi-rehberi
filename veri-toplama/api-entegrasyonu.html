API tasarımı Standartlaştırma Version yönetimi Hata yönetimi

Kullanım Amacı: İyi tasarlanmış, sürdürülebilir ve kullanıcı dostu API'ler oluşturmak. Örnek: Richardson Maturity Model kullanarak REST API'nin olgunluk seviyesini belirlemek.

💡 Ne Zaman Kullanılır?

Seviye Detayları

  • Level 0 - POX over HTTP
    • Tek endpoint
    • HTTP POST kullanımı
    • SOAP benzeri
  • Level 1 - Resources
    • Çoklu endpoint
    • Resource-based URL'ler
    • HTTP POST dominant
  • Level 2 - HTTP Verbs
    • HTTP metodları (GET, POST, PUT, DELETE)
    • HTTP status kodları
    • Standart REST
  • Level 3 - HATEOAS
    • Hypermedia kontrolleri
    • Keşfedilebilirlik
    • Self-documenting

Authentication and Security

OAuth 2.0, JWT token management ve API güvenlik stratejileri

🚀 Ne İşe Yarar?

Kimlik doğrulama Yetkilendirme Güvenlik Rate limiting

Kullanım Amacı: API'leri yetkisiz erişimden korumak, kullanıcı kimlik doğrulaması yapmak ve güvenli veri transferi sağlamak. Örnek: OAuth 2.0 ile üçüncü parti uygulamalara sınırlı erişim izni vermek.

💡 Ne Zaman Kullanılır?

Üçüncü parti uygulamalara sınırlı erişim izni vermek, kullanıcı kimlik doğrulaması yapmak ve güvenli API erişimi sağlamak için.

OAuth 2.0 Flow Türleri

  • Authorization Code
    • Web uygulamaları için
    • En güvenli flow
    • Server-side authentication
  • Implicit
    • Single-page apps için
    • Daha az güvenli
    • Client-side authentication
  • Resource Owner Password Credentials
    • Yüksek güven gerektiren durumlar
    • First-party apps
    • Kullanıcı adı/şifre ile
  • Client Credentials
    • Machine-to-machine
    • Server-to-server
    • App authentication

Authorization Code Flow

1. Authorization Request

Kullanıcıyı authorization server'a yönlendir

2. User Authorization

Kullanıcı izin verir ve authorization code döner

3. Token Exchange

Authorization code'u access token ile değiştir

4. API Access

Access token ile API'ye istek at

OAuth 2.0 Authorization Code Flow Demo
// OAuth 2.0 Authorization Code Flow - Node.js implementasyonu
const express = require('express');
const axios = require('axios');
const querystring = require('querystring');

const app = express();

// OAuth configuration
const OAUTH_CONFIG = {
  clientId: 'your-client-id',
  clientSecret: 'your-client-secret',
  authorizationEndpoint: 'https://provider.com/oauth/authorize',
  tokenEndpoint: 'https://provider.com/oauth/token',
  redirectUri: 'http://localhost:3000/callback',
  scopes: ['read:user', 'read:email']
};

// 1. Authorization Request
app.get('/auth', (req, res) => {
  const authUrl = `${OAUTH_CONFIG.authorizationEndpoint}?${querystring.stringify({
    client_id: OAUTH_CONFIG.clientId,
    redirect_uri: OAUTH_CONFIG.redirectUri,
    response_type: 'code',
    scope: OAUTH_CONFIG.scopes.join(' '),
    state: generateRandomString() // CSRF koruması
  })}`;
  
  res.redirect(authUrl);
});

// 2. Callback Handler - Token Exchange
app.get('/callback', async (req, res) => {
  const { code, state } = req.query;
  
  try {
    // State validation (CSRF koruması)
    if (!validateState(state)) {
      return res.status(400).send('Invalid state parameter');
    }
    
    // 3. Token Exchange
    const tokenResponse = await axios.post(OAUTH_CONFIG.tokenEndpoint,
      querystring.stringify({
        client_id: OAUTH_CONFIG.clientId,
        client_secret: OAUTH_CONFIG.clientSecret,
        code: code,
        redirect_uri: OAUTH_CONFIG.redirectUri,
        grant_type: 'authorization_code'
      }), {
        headers: {
          'Content-Type': 'application/x-www-form-urlencoded'
        }
      }
    );
    
    const { access_token, refresh_token, expires_in } = tokenResponse.data;
    
    // 4. API Access with Access Token
    const userResponse = await axios.get('https://api.provider.com/user', {
      headers: {
        'Authorization': `Bearer ${access_token}`
      }
    });
    
    // Kullanıcı bilgilerini işle
    res.json({
      user: userResponse.data,
      tokens: {
        access_token,
        refresh_token,
        expires_in
      }
    });
    
  } catch (error) {
    console.error('OAuth error:', error.response?.data || error.message);
    res.status(500).send('Authentication failed');
  }
});

// Yardımcı fonksiyonlar
function generateRandomString() {
  return Math.random().toString(36).substring(2, 15);
}

function validateState(state) {
  // State validation logic (session vs.)
  return true; // Basitleştirilmiş
}

// Token refresh endpoint
app.post('/refresh', async (req, res) => {
  const { refresh_token } = req.body;
  
  try {
    const refreshResponse = await axios.post(OAUTH_CONFIG.tokenEndpoint,
      querystring.stringify({
        client_id: OAUTH_CONFIG.clientId,
        client_secret: OAUTH_CONFIG.clientSecret,
        refresh_token: refresh_token,
        grant_type: 'refresh_token'
      })
    );
    
    res.json(refreshResponse.data);
  } catch (error) {
    res.status(400).json({ error: 'Token refresh failed' });
  }
});

// GraphQL Schema Tanımı
const typeDefs = `
  type User {
    id: ID!
    name: String!
    email: String!
    posts: [Post!]!
  }

  type Post {
    id: ID!
    title: String!
    content: String!
    author: User!
    comments: [Comment!]!
  }

  type Comment {
    id: ID!
    text: String!
    author: User!
    post: Post!
  }

  type Query {
    users: [User!]!
    user(id: ID!): User
    posts: [Post!]!
    post(id: ID!): Post
  }

  type Mutation {
    createUser(name: String!, email: String!): User!
    createPost(title: String!, content: String!, authorId: ID!): Post!
    addComment(text: String!, authorId: ID!, postId: ID!): Comment!
  }

  type Subscription {
    newPost: Post!
    newComment(postId: ID!): Comment!
  }
`;

// GraphQL Resolver'ları
const resolvers = {
  Query: {
    users: () => users,
    user: (parent, { id }) => users.find(user => user.id === id),
    posts: () => posts,
    post: (parent, { id }) => posts.find(post => post.id === id),
  },
  User: {
    posts: (parent) => posts.filter(post => post.authorId === parent.id),
  },
  Post: {
    author: (parent) => users.find(user => user.id === parent.authorId),
    comments: (parent) => comments.filter(comment => comment.postId === parent.id),
  },
  Comment: {
    author: (parent) => users.find(user => user.id === parent.authorId),
    post: (parent) => posts.find(post => post.id === parent.postId),
  },
  Mutation: {
    createUser: (parent, { name, email }) => {
      const user = { id: String(users.length + 1), name, email };
      users.push(user);
      return user;
    },
    createPost: (parent, { title, content, authorId }) => {
      const post = { id: String(posts.length + 1), title, content, authorId };
      posts.push(post);
      pubsub.publish('NEW_POST', { newPost: post });
      return post;
    },
    addComment: (parent, { text, authorId, postId }) => {
      const comment = { id: String(comments.length + 1), text, authorId, postId };
      comments.push(comment);
      pubsub.publish('NEW_COMMENT', { newComment: comment, postId });
      return comment;
    },
  },
  Subscription: {
    newPost: {
      subscribe: () => pubsub.asyncIterator(['NEW_POST']),
    },
    newComment: {
      subscribe: (parent, { postId }) => pubsub.asyncIterator(`NEW_COMMENT_${postId}`),
    },
  },
};

// GraphQL Query Örnekleri

          text

Performance Optimization

Rate limiting, caching, circuit breaker ve diğer performans optimizasyon teknikleri

🚀 Ne İşe Yarar?

Rate limiting Caching Circuit breaker Load balancing

Kullanım Amacı: API performansını artırmak, kaynak kullanımını optimize etmek ve sistem stabilitesini sağlamak. Örnek: Rate limiting ile API'yi kötüye kullanımı önlemek.

💡 Ne Zaman Kullanılır?

API'yi kötüye kullanımı önlemek, sunucu yükünü kontrol etmek ve adil kullanım sağlamak için.

Rate Limiting Algoritmaları

  • Fixed Window
    • Sabit zaman aralığı
    • Basit implementasyon
    • Burst problem
  • Sliding Window
    • Hareketli zaman aralığı
    • Daha adil
    • Karmaşık implementasyon
  • Token Bucket
    • Token bazlı
    • Burst destek
    • Esnek kullanım
  • Leaky Bucket
    • Sabit çıkış hızı
    • Smooth traffic
    • Queue-based

Rate Limiting Stratejileri

IP-based Limiting

• Client IP adresine göre
• Basit ve etkili
• VPN bypass riski

User-based Limiting

• Kullanıcı kimliğine göre
• Daha adil
• Authentication gerektirir

Endpoint-based Limiting

• Endpoint'e göre farklı limitler
• Resource-based
• Fine-grained control

Redis ile Rate Limiting Demo
// Redis ile Sliding Window Rate Limiting
const redis = require('redis');
const client = redis.createClient();

class RateLimiter {
  constructor({ windowSize, maxRequests }) {
    this.windowSize = windowSize; // saniye cinsinden
    this.maxRequests = maxRequests;
  }

  async checkRateLimit(identifier) {
    const key = `rate_limit:${identifier}`;
    const now = Date.now();
    const windowStart = now - (this.windowSize * 1000);

    try {
      // Eski kayıtları temizle
      await client.zremrangebyscore(key, 0, windowStart);
      
      // Mevcut request sayısını al
      const requestCount = await client.zcard(key);
      
      if (requestCount >= this.maxRequests) {
        return {
          allowed: false,
          remaining: 0,
          retryAfter: await this.getRetryAfter(key, windowStart)
        };
      }

      // Yeni request'i ekle
      await client.zadd(key, now, now.toString());
      await client.expire(key, this.windowSize);

      return {
        allowed: true,
        remaining: this.maxRequests - requestCount - 1,
        retryAfter: null
      };
      
    } catch (error) {
      console.error('Rate limit error:', error);
      // Hata durumunda rate limiting'i bypass et
      return { allowed: true, remaining: this.maxRequests, retryAfter: null };
    }
  }

  async getRetryAfter(key, windowStart) {
    const oldestRequest = await client.zrange(key, 0, 0, 'WITHSCORES');
    if (oldestRequest.length > 0) {
      const oldestTimestamp = parseInt(oldestRequest[1]);
      return Math.ceil((oldestTimestamp + (this.windowSize * 1000) - Date.now()) / 1000);
    }
    return this.windowSize;
  }
}

// Rate Limiting Middleware
const rateLimiter = new RateLimiter({
  windowSize: 60, // 60 saniye
  maxRequests: 100 // maksimum 100 istek
});

app.use(async (req, res, next) => {
  const identifier = req.ip; // IP bazlı rate limiting
  
  const limitResult = await rateLimiter.checkRateLimit(identifier);
  
  if (!limitResult.allowed) {
    res.set({
      'X-RateLimit-Limit': rateLimiter.maxRequests,
      'X-RateLimit-Remaining': limitResult.remaining,
      'X-RateLimit-Reset': limitResult.retryAfter,
      'Retry-After': limitResult.retryAfter
    });
    return res.status(429).json({
      error: 'Too Many Requests',
      message: `Rate limit exceeded. Try again in ${limitResult.retryAfter} seconds.`,
      retryAfter: limitResult.retryAfter
    });
  }

  res.set({
    'X-RateLimit-Limit': rateLimiter.maxRequests,
    'X-RateLimit-Remaining': limitResult.remaining,
    'X-RateLimit-Reset': limitResult.retryAfter || rateLimiter.windowSize
  });

  next();
});

// Farklı endpoint'ler için farklı rate limit'ler
const endpointLimits = {
  '/api/auth': { windowSize: 300, maxRequests: 5 }, // Login için daha sıkı
  '/api/users': { windowSize: 60, maxRequests: 100 },
  '/api/admin': { windowSize: 60, maxRequests: 1000 } // Admin için yüksek limit
};
